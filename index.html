Зачем нужны модули?

  • Пространство имён
  • Повторное использование
  • Устранение связей (Decoupling)

Использование функций в качестве пространств имён


var items = ['one', 'two', 'three'];

function getItem(n) {
  return items(n);
}

getItem(1)                           // 'two'
items                                // ['one', 'two', 'three']
					

var getItem = (function() {
  var items = ['one', 'two', 'three'];

  return function(n) {
    return items(n);
  }
})

getItem(1)                           // 'two'
items                                // items is not defined

Объекты в качестве интерфейсов


var Store = (function() {
  var items = [];

  return {
    addItem: function(item) {
      items.push(item);
    },

    getItem: function(n) {
      return items[n];
    }
  }
})();

Store.addItem('ho')
Store.getItem(0)                      // 'ho'

Вариации на тему


var Store = (function() {
  var items = [];

  function addItem(item) {
    items.push(item);
  }

  function getItem(item) {
    items.push(item);
  }

  return {
    addItem: addItem
  }
})();

Store.addItem('ho')
Store.getItem(0)           // undefined is not a function

шаг 1


(function(exports) {
  var items = [];

  exports.addItem = function(item) {
    items.push(item);
  };

  exports.getItem = function(n) {
    return items[n]
  };
})(this.Store = {});

Store.addItem('ho')
Store.getItem(0)                      // 'ho'

Eval vs Function constructor


function evalAndReturnX(code) {
  eval(code);
  return x;
}

evalAndReturnX("var x = 2");          // 2
					

var square = new Function("n", "return n * n;");
square(4)                                        // 16

шаг 2. require


function require(name) {
  var code = new Function("exports", readFile(name));
  var exports = {};
  code(exports);
  return exports;
}

require v2


function require(name) {
  if (name in require.cache)
    return require.cache[name];

  var code = new Function("exports, module", readFile(name));
  var exports = {}, module = {exports: exports};
  code(exports, module);

  require.cache[name] = module.exports;
  return module.exports;
}
require.cache = Object.create(null);
